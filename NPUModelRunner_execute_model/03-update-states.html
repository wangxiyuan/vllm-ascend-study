03 - _update_states 状态更新

vllm/vllm/v1/worker/gpu_model_runner.py:934-1133

方法概述

_update_states 方法负责根据调度器输出更新缓存状态和持久批处理数据。 这是 _prepare_inputs 函数的前置步骤，确保输入 GPU 张量正确创建。 

def _update_states(self, scheduler_output: "SchedulerOutput") -> None:
    """Update the cached states and the persistent batch with the scheduler
    output.

    The updated states are used by the `_prepare_inputs` function to create
    the input GPU tensors for the model.

    The SamplingMetadata is updated and copied to the GPU if there is a
    new/resumed/paused/finished request in the batch.
    """

核心职责

┌─────────────────────────────────────────────────────────┐ │ _update_states 职责 │ ├─────────────────────────────────────────────────────────┤ │ 1. 移除已完成请求的缓存状态 │ │ 2. 从持久批中移除完成的请求 │ │ 3. 释放编码器缓存 │ │ 4. 移除未调度的请求 │ │ 5. 添加新请求到缓存 │ │ 6. 更新运行/恢复请求的状态 │ │ 7. 将请求添加到持久批 │ └─────────────────────────────────────────────────────────┘

处理流程

1. 移除已完成请求

# 从缓存状态中移除完成的请求
for req_id in scheduler_output.finished_req_ids:
    self.requests.pop(req_id, None)
    self.num_prompt_logprobs.pop(req_id, None)

# 从持久批中移除完成的请求
for req_id in scheduler_output.finished_req_ids:
    self.input_batch.remove_request(req_id)

边界情况处理

finished_req_ids 和 scheduled_req_ids 可能重叠。 这发生在请求被中止然后以相同 ID 重新提交的情况。此时将它们视为两个独立的请求： 清除第一个请求的缓存状态，并将第二个作为新请求处理。

2. 释放编码器缓存

# 释放缓存的编码器输出（用于多模态模型）
for mm_hash in scheduler_output.free_encoder_mm_hashes:
    self.encoder_cache.pop(mm_hash, None)

3. 移除未调度的请求

# 计算未调度的请求
scheduled_req_ids = scheduler_output.num_scheduled_tokens.keys()
cached_req_ids = self.input_batch.req_id_to_index.keys()
resumed_req_ids = scheduler_output.scheduled_cached_reqs.resumed_req_ids

unscheduled_req_ids = cached_req_ids - (scheduled_req_ids - resumed_req_ids)

# 从持久批中移除未调度的请求（但保留缓存状态）
for req_id in unscheduled_req_ids:
    self.input_batch.remove_request(req_id)

持久批优化假设

持久批优化假设连续的批次包含大部分相同的请求。如果批次的请求重叠率低 （例如在两组不同的请求之间交替），此优化会变得非常低效。

4. 添加新请求

reqs_to_add: list[CachedRequestState] = []

for new_req_data in scheduler_output.scheduled_new_reqs:
    req_id = new_req_data.req_id

    # 处理流式请求（可能已存在）
    if req_id in self.requests:
        req_state = self._update_streaming_request(req_id, new_req_data)
        reqs_to_add.append(req_state)
        continue

    # 创建随机数生成器（用于确定性采样）
    sampling_params = new_req_data.sampling_params
    if sampling_params and sampling_params.sampling_type == SamplingType.RANDOM_SEED:
        generator = torch.Generator(device=self.device)
        generator.manual_seed(sampling_params.seed)
    else:
        generator = None

    # 创建请求状态
    req_state = CachedRequestState(
        req_id=req_id,
        prompt_token_ids=new_req_data.prompt_token_ids,
        prompt_embeds=new_req_data.prompt_embeds,
        mm_features=new_req_data.mm_features,
        sampling_params=sampling_params,
        pooling_params=pooling_params,
        generator=generator,
        block_ids=new_req_data.block_ids,
        num_computed_tokens=new_req_data.num_computed_tokens,
        output_token_ids=[],
        lora_request=new_req_data.lora_request,
    )
    self.requests[req_id] = req_state
    reqs_to_add.append(req_state)

5. 更新运行/恢复请求状态

is_last_rank = get_pp_group().is_last_rank
req_data = scheduler_output.scheduled_cached_reqs
scheduled_spec_tokens = scheduler_output.scheduled_spec_decode_tokens

    req_index = self.input_batch.req_id_to_index.get(req_id)

    # 处理异步调度 + 投机解码
    if req_state.prev_num_draft_len and self.use_async_scheduling:
        # 更新 num_computed_tokens 以反映接受的投机令牌
        ...

    # 更新已计算令牌数
    req_state.num_computed_tokens = num_computed_tokens

    # 更新输出令牌
    if not is_last_rank:
        # 流水线并行：非最后阶段
        new_token_ids = req_data.new_token_ids[i]
        ...

    # 更新块 ID
    if not resumed_from_preemption:

6. 将请求添加到持久批

for req_state in reqs_to_add:
    self.input_batch.add_request(req_state)

数据结构

CachedRequestState

@dataclass
class CachedRequestState:
    req_id: str
    prompt_token_ids: list[int]
    prompt_embeds: torch.Tensor | None
    mm_features: list[MultiModalFeature]
    sampling_params: SamplingParams | None
    pooling_params: PoolingParams | None
    generator: torch.Generator | None
    block_ids: list[list[int]]
    num_computed_tokens: int
    output_token_ids: list[int]
    lora_request: LoRARequest | None

    @property
    def num_tokens(self) -> int:
        return len(self.prompt_token_ids) + len(self.output_token_ids)
