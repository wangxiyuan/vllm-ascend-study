2.3 完整源码分析

def initialize_kv_cache(self, kv_cache_config: KVCacheConfig) -> None:
    # 1. 深拷贝配置，避免修改原始配置对象
    kv_cache_config = deepcopy(kv_cache_config)
    self.kv_cache_config = kv_cache_config

    # 2. 添加编码器专用层到 KV Cache 配置
    self.may_add_encoder_only_layers_to_kv_cache_config()

    # 3. 添加 KV 共享层到 KV Cache 组
    self.maybe_add_kv_sharing_layers_to_kv_cache_groups(kv_cache_config)

    # 4. 初始化 Attention Backend（必须在使用 attn_groups 之前）
    # NOTE(cmq): initialize_attn_backend must before using self.attn_groups
    self.initialize_attn_backend(kv_cache_config)

    # 5. 设置混合块标志：当有多个 Attention 组时使用混合块
    self.use_hybrid_blocks = len(self.attn_groups) > 1

    # 6. 检测是否需要 num_accepted_tokens（通过 MambaSpec 判断）
    # NOTE: Currently, we determine whether we need `num_accepted_tokens`
    # through `MambaSpec`.
    self.need_accepted_tokens = any(
        [isinstance(attn_group[0].kv_cache_spec, MambaSpec)
         for attn_group in self.attn_groups]
    )

    # 7. 重新初始化输入批次（如果 block sizes 不同）
    self.may_reinitialize_input_batch(kv_cache_config)

    # 8. 初始化 KV Cache 张量（核心分配逻辑）
    kv_caches = self.initialize_kv_cache_tensors(kv_cache_config)

    # 9. 注册 KV Cache 到 KV Transfer 组（如果启用）
    if has_kv_transfer_group():
        get_kv_transfer_group().register_kv_caches(kv_caches)

    # 10. 初始化路由专家捕获器（如果启用）
    if self.model_config.enable_return_routed_experts:
        self.init_routed_experts_capturer()

关键注释

NOTE(cmq): initialize_attn_backend 必须在任何使用 self.attn_groups 的操作之前调用。这是因为 attn_groups 是在该方法中被初始化的，后续的所有操作都依赖于这个数据结构。

2.4 initialize_attn_backend 详解

initialize_attn_backend 方法负责创建和管理所有 Attention Backend 实例。在 vLLM 中，不同的层可能使用不同的注意力实现，因此需要为每种类型创建相应的后端。

def initialize_attn_backend(self, kv_cache_config: KVCacheConfig) -> None:
    """
    Initialize the attention backends and attention metadata builders.
    """
    # 断言检查：确保 Attention Backend 尚未初始化
    assert len(self.attn_groups) == 0, "Attention backends are already initialized"

    # 定义 AttentionGroupKey 命名元组用于去重
    class AttentionGroupKey(NamedTuple):
        attn_backend: type[AttentionBackend]
        kv_cache_spec: KVCacheSpec

AttentionGroupKey 数据结构

AttentionGroupKey 是一个局部定义的命名元组，用于对 Attention Backend 进行去重和分组。它将相同类型的 Backend 归为一组，以便统一管理和优化。

2.5 get_attn_backends_for_group 函数

这是一个内部辅助函数，用于从配置中提取所有唯一的 Attention Backend 类型。

def get_attn_backends_for_group(
    kv_cache_group_spec: KVCacheGroupSpec,
) -> tuple[dict[AttentionGroupKey, list[str]], set[type[AttentionBackend]]]:
    # 从 vllm_config 中获取指定层名的 AttentionLayerBase 实例
    layers = get_layers_from_vllm_config(
        self.vllm_config,
        AttentionLayerBase,
        kv_cache_group_spec.layer_names
    )

    attn_backends = {}
    attn_backend_layers = defaultdict(list)

    # 遍历所有层，收集注意力后端
    for layer_name in kv_cache_group_spec.layer_names:
        # 从层实例获取其关联的注意力后端
        attn_backend = layers[layer_name].get_attn_backend()

        # 使用全类名作为键（支持动态创建的子类）
        full_cls_name = attn_backend.full_cls_name()

        # 处理 UniformTypeKVCacheSpecs 类型
        layer_kv_cache_spec = kv_cache_group_spec.kv_cache_spec
        if isinstance(layer_kv_cache_spec, UniformTypeKVCacheSpecs):
            layer_kv_cache_spec = layer_kv_cache_spec.kv_cache_specs[layer_name]

        # 创建唯一键
        key = (full_cls_name, layer_kv_cache_spec)
        attn_backends[key] = AttentionGroupKey(attn_backend, layer_kv_cache_spec)
        attn_backend_layers[key].append(layer_name)

    # 返回映射关系和后端类型集合
    return (
        {attn_backends[k]: v for k, v in attn_backend_layers.items()},
        set(group_key.attn_backend for group_key in attn_backends.values())
    )

2.6 create_attn_groups 函数

根据获取的后端映射关系创建 AttentionGroup 实例。

def create_attn_groups(
    attn_backends_map: dict[AttentionBackend, list[str]],
    kv_cache_group_id: int
) -> list[AttentionGroup]:
    attn_groups: list[AttentionGroup] = []

    for (attn_backend, kv_cache_spec), layer_names in attn_backends_map.items():
        # 为每组创建 AttentionMetadataBuilder
        attn_metadata_builders = []
        attn_metadata_builders.append(
            attn_backend.get_builder_cls()(
                kv_cache_spec,
                layer_names,
                self.vllm_config,
                self.device,
            )
        )

        # 创建 AttentionGroup
        attn_group = AttentionGroup(
            attn_backend,
            layer_names,
            kv_cache_spec,
            kv_cache_group_id,
            attn_metadata_builders
        )
        attn_groups.append(attn_group)

    return attn_groups

为什么需要 AttentionGroup？

在 vLLM 中，不同的模型层可能使用不同的注意力机制实现：

  • 标准 Attention：用于大多数 Transformer 层
  • MLA（Multi-Latent Attention）：用于 DeepSeek 等模型
  • Sliding Window Attention：用于局部注意力场景
  • Linear Attention：用于 Mamba 等状态空间模型

AttentionGroup 将相同类型的层分组，便于统一管理和优化。

2.9 calculate_reorder_batch_threshold

该方法检查所有 Attention Backend 是否需要批次重排序，并确保所有后端的阈值设置一致。

def calculate_reorder_batch_threshold(self) -> None:
    """
    Check that if any backends reorder batches; that the reordering
    is compatible (e.g., decode threshold is the same)
    """
    for group in self._attn_group_iterator():
        attn_metadata_builder_i = group.get_metadata_builder()

        if hasattr(attn_metadata_builder_i, "reorder_batch_threshold"):
            reorder_batch_threshold_i = attn_metadata_builder_i.reorder_batch_threshold

            if reorder_batch_threshold_i is not None:
                if self.reorder_batch_threshold is not None:
                    # 检查阈值是否一致
                    if reorder_batch_threshold_i != self.reorder_batch_threshold:
                        raise ValueError(
                            f"Attention backend reorders decodes with threshold "
                            f"{reorder_batch_threshold_i} but other backend uses "
                            f"threshold {self.reorder_batch_threshold}"
                        )
                else:
                    self.reorder_batch_threshold = reorder_batch_threshold_i
