配置参数

SpeculativeConfig

位置: vllm/config/speculative.py

核心配置项

@config
class SpeculativeConfig:
    # 基本控制
    num_speculative_tokens: int  # 推测 token 数量
    method: SpeculativeMethod   # 方法类型
    model: str | None           # draft 模型路径

    # Ngram 配置
    prompt_lookup_min: int  # n-gram 最小长度
    prompt_lookup_max: int  # n-gram 最大长度

    # Suffix Decoding 配置
    suffix_decoding_max_tree_depth: int
    suffix_decoding_max_cached_requests: int
    suffix_decoding_max_spec_factor: float
    suffix_decoding_min_token_prob: float

    # 高级控制
    parallel_drafting: bool          # 并行草稿
    disable_padded_drafter_batch: bool
    use_local_argmax_reduction: bool

基本配置示例

speculative_config = {
    # 必须参数
    "method": "eagle",                    # 方法类型
    "num_speculative_tokens": 4,          # 推测 token 数量

    # 方法相关参数
    "model": "yuhuili/EAGLE-LLaMA3-8B",   # draft 模型路径 (某些方法需要)
    "prompt_lookup_max": 5,               # ngram 最大长度
    "prompt_lookup_min": 2,               # ngram 最小长度

    # 高级选项
    "parallel_drafting": False,           # 并行草稿
    "enforce_eager": False,               # 禁用 CUDA Graph
    "draft_tensor_parallel_size": 1,      # draft 模型 TP 大小
}

性能调优建议

参数 影响 推荐值
num_speculative_tokens 越大加速越多，但接受率下降 2-5 (EAGLE), 4-8 (MTP)
prompt_lookup_max 影响 ngram 匹配质量 5-10
draft_tensor_parallel_size draft 模型并行度 1 或与 target 相同
suffix_decoding_max_tree_depth 后缀树深度，影响匹配长度 16-32

使用示例

命令行使用

# EAGLE 方法
python examples/offline_inference/spec_decode.py \
    --method eagle \
    --num-spec-tokens 3 \
    --model-dir meta-llama/Llama-3.1-8B-Instruct \
    --eagle-dir yuhuili/EAGLE-LLaMA3.1-Instruct-8B

# Ngram 方法
python examples/offline_inference/spec_decode.py \
    --method ngram \
    --num-spec-tokens 4 \
    --prompt-lookup-max 5 \
    --prompt-lookup-min 2

# MTP 方法 (DeepSeek)
python examples/offline_inference/spec_decode.py \
    --method mtp \
    --num-spec-tokens 4 \
    --model-dir deepseek-ai/DeepSeek-V3

Python API 使用

from vllm import LLM, SamplingParams

# 配置 speculative decoding
llm = LLM(
    model="meta-llama/Llama-3.1-8B-Instruct",
    speculative_config={
        "method": "eagle",
        "model": "yuhuili/EAGLE-LLaMA3.1-Instruct-8B",
        "num_speculative_tokens": 3,
    },
)

sampling_params = SamplingParams(
    temperature=0.7,
    max_tokens=256,
)

prompts = ["Hello, my name is"]
outputs = llm.generate(prompts, sampling_params)

for output in outputs:
    print(output.outputs[0].text)

获取 Spec Decode 指标

from vllm.v1.metrics.reader import Counter, Vector

metrics = llm.get_metrics()

num_drafts = 0
num_draft_tokens = 0
num_accepted_tokens = 0

for metric in metrics:
    if metric.name == "vllm:spec_decode_num_drafts":
        num_drafts = metric.value
    elif metric.name == "vllm:spec_decode_num_draft_tokens":
        num_draft_tokens = metric.value
    elif metric.name == "vllm:spec_decode_num_accepted_tokens":
        num_accepted_tokens = metric.value

# 计算接受率
acceptance_rate = num_accepted_tokens / num_draft_tokens * 100
mean_acceptance_length = 1 + (num_accepted_tokens / num_drafts)

print(f"Acceptance Rate: {acceptance_rate:.1f}%")
print(f"Mean Acceptance Length: {mean_acceptance_length:.2f}")
