核心数据结构

class Request:
    self.spec_token_ids: list[int] = []  # 存储 draft model 生成的待验证 tokens
    self.num_computed_tokens = 0  # 已计算的 token 数
    self.num_tokens = len(prompt) + len(output)  # prompt + output tokens

@dataclass
class SchedulerOutput:
    scheduled_spec_decode_tokens: dict[str, list[int]]  # req_id -> [draft_token_1, draft_token_2, ...]

Scheduler 调度流程

def schedule(self) -> SchedulerOutput:
    scheduled_spec_decode_tokens: dict[str, list[int]] = {}

    for request in scheduled_running_reqs:
        # ... (省略 block 分配等逻辑)

        # ============= Speculative Decode 相关 =============
        if request.spec_token_ids:
            # 计算需要调度的 spec token 数量
            num_scheduled_spec_tokens = (
                num_new_tokens           # 本轮调度的新 token 数
                + request.num_computed_tokens  # 已计算的 token 数
                - request.num_tokens         # prompt + output tokens
                - request.num_output_placeholders  # 异步调度中的占位符
            )

            if num_scheduled_spec_tokens > 0:
                # 裁剪 spec_token_ids 到正确长度
                spec_token_ids = request.spec_token_ids
                if len(spec_token_ids) > num_scheduled_spec_tokens:
                    spec_token_ids = spec_token_ids[:num_scheduled_spec_tokens]

                # 添加到 scheduled_spec_decode_tokens
                scheduled_spec_decode_tokens[request.request_id] = spec_token_ids

            # 清空 request 的 spec_token_ids（将在下一轮被更新）
            request.spec_token_ids = []

    return SchedulerOutput(
        scheduled_spec_decode_tokens=scheduled_spec_decode_tokens,
        ...
    )

2.2 Target Model 前向传播

def execute_model(self, scheduler_output: "SchedulerOutput", ...) -> ModelRunnerOutput | None:
    # 1. _prepare_inputs() 构建 SpecDecodeMetadata
    logits_indices, spec_decode_metadata, ... = self._prepare_inputs(...)

    # 2. Target Model 前向传播
    with set_ascend_forward_context(...):
        hidden_states = self._model_forward(...)

    # 3. 计算 logits（只在需要的位置）
    sample_hidden_states = hidden_states[logits_indices]
    logits = self.model.compute_logits(sample_hidden_states)

    # 4. 保存状态到 ExecuteModelState（传递给 sample_tokens）
    self.execute_model_state = ExecuteModelState(
        scheduler_output=scheduler_output,
        logits=logits,
        spec_decode_metadata=spec_decode_metadata,
        spec_decode_common_attn_metadata=spec_decode_common_attn_metadata,
        hidden_states=hidden_states,
        sample_hidden_states=sample_hidden_states,
        aux_hidden_states=aux_hidden_states,
        attn_metadata=attn_metadata,
        positions=positions,
        ec_connector_output=ec_connector_output,
        cudagraph_stats=cudagraph_stats,
    )

    return None  # 返回 None，等待 sample_tokens 调用

3.2 Rejection Sampling 核心逻辑

def _sample(self, logits, spec_decode_metadata):
    sampling_metadata = self.input_batch.sampling_metadata

    if spec_decode_metadata is None:
        # 普通采样（无 spec decode）
        return self.sampler(logits=logits, sampling_metadata=sampling_metadata)

    # Spec decode 采样：验证 draft token
    if lmhead_tp_enable() and logits is not None:
        logits = logits[: len(spec_decode_metadata.logits_indices)]

    sampler_output = self.rejection_sampler(
        spec_decode_metadata,
        None,  # draft_probs
        logits,       # target model 的 logits [num_tokens + batch_size, vocab_size]
        sampling_metadata,
    )
    return sampler_output

位置: vllm/v1/core/sched/scheduler.py:1297-1345

4.1 update_from_output：更新 request 状态

def update_from_output(
    self,
    scheduler_output: SchedulerOutput,
    model_runner_output: ModelRunnerOutput,
) -> dict[int, EngineCoreOutputs]:
    sampled_token_ids = model_runner_output.sampled_token_ids

    for req_id, request in self.requests.items():
        if request.is_finished():
            continue

        req_index = model_runner_output.req_id_to_index[req_id]
        generated_token_ids = sampled_token_ids[req_index] if sampled_token_ids else []

        # ============= Speculative Decode 状态更新 =============
        scheduled_spec_token_ids = scheduler_output.scheduled_spec_decode_tokens.get(req_id)

        if scheduled_spec_token_ids and generated_token_ids:
            num_draft_tokens = len(scheduled_spec_token_ids)
            num_accepted = len(generated_token_ids) - 1  # 最后 1 个是 bonus token
            num_rejected = num_draft_tokens - num_accepted

            # num_computed_tokens 回退（因为 reject 的 token 需要重新计算）
            if request.num_computed_tokens > 0:
                request.num_computed_tokens -= num_rejected

            # 异步调度时，num_output_placeholders 也需要回退
            if request.num_output_placeholders > 0:
                request.num_output_placeholders -= num_rejected

        # 更新 request 的 token 列表
        if generated_token_ids:
            request.append_output_token_ids(generated_token_ids)

4.2 update_draft_token_ids：更新下一轮的 spec tokens

def update_draft_token_ids(self, draft_token_ids: DraftTokenIds) -> None:
    """
    在 model_runner 生成新的 draft tokens 后，更新到 scheduler 的 request 中
    """
    for req_id, spec_token_ids in zip(
        draft_token_ids.req_ids,
        draft_token_ids.draft_token_ids,
    ):
        request = self.requests.get(req_id)
        if request is None or request.is_finished():
            continue

        if request.is_prefill_chunk:
            # Prefill chunk 阶段忽略 draft tokens
            if request.spec_token_ids:
                request.spec_token_ids = []
            continue

        # 应用 grammar 验证（结构化输出场景）
        if self.structured_output_manager.should_advance(request):
            metadata = request.structured_output_request
            spec_token_ids = metadata.grammar.validate_tokens(spec_token_ids)

        # 更新 request 的 spec_token_ids
        request.spec_token_ids = spec_token_ids
